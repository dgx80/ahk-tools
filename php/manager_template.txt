    /** @var \Sup_timeslot_model $timeslotModel */
    private static $timeslotModel;

    ////////////////////////////////////////////////////////////REGION////////////////////////////////////////////
    //////////////////////////////////////////////////////////////////////////////////////////////////////////////
    //////////////////////////////
    //////////////////////////////          Constructor
    //////////////////////////////
    //////////////////////////////////////////////////////////////////////////////////////////////////////////////
    //////////////////////////////////////////////////////////////////////////////////////////////////////////////

    public function __construct()
    {
        if (!self::$timeslotModel) {
            self::$timeslotModel = new \Sup_timeslot_model();
        }
    }

    ////////////////////////////////////////////////////////////REGION////////////////////////////////////////////
    //////////////////////////////////////////////////////////////////////////////////////////////////////////////
    //////////////////////////////
    //////////////////////////////          Accessor
    //////////////////////////////
    //////////////////////////////////////////////////////////////////////////////////////////////////////////////
    //////////////////////////////////////////////////////////////////////////////////////////////////////////////

    ////////////////////////////////////////////////////////////////////////////////
    //////////////////////////////Getter////////////////////////////////////////////
    ////////////////////////////////////////////////////////////////////////////////

    public function getTimeslot($id)
    {
        return self::$timeslotModel->get_one($id);
    }
